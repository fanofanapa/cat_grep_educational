#include "my_cat_func.h"

int main(int argc, char **argv) {
  if (argc <= 1) {
    printf("usage: my_cat <optional: flags> <filenames> ...");
    return -1;
  }

  Flags flags = {0};
  int flag;
  while ((flag = getopt(argc, argv, POSSIBLE_FLAGS)) != -1) {
    if (flag == 'b') {
      flags.b = 1;
    } else if (flag == 'e') {
      flags.e = 1;
    } else if (flag == 'n') {
      flags.n = 1;
    } else if (flag == 's') {
      flags.s = 1;
    } else if (flag == 't') {
      flags.t = 1;
    }
  }

  int counter_for_lines = 1;
  for (int i = optind; i < argc; i++) {
    if (is_file_exists(argv[i])) {
      print_file(argv[i], &flags, &counter_for_lines);
    } else {
      printf("my_cat: %s No such file or directory\n", argv[i]);
      return -1;
    }
  }
}
